Quick Example

Start Building in 5 Minutes

// Initialize the lvlBase client const { LvlBase } = require('@lvlbase/sdk'); const client = new LvlBase({ apiKey: process.env.LVLBASE_API_KEY, orgId: process.env.LVLBASE_ORG_ID }); // List all users in your organization const users = await client.users.list({ page: 1, limit: 50, filter: { department: 'Engineering' } }); // Get AI insights for your organization const insights = await client.ai.getInsights({ type: 'engagement', period: 'last_30_days' }); // Create a new guild const guild = await client.guilds.create({ name: 'Alpha Squad', members: ['usr_001', 'usr_002', 'usr_003'], objectives: ['Complete onboarding module', 'Hit Q1 targets'] }); console.log(`Guild created: ${guild.id}`);
